\begin{frame}
	\frametitle{Основные направления в развитии нейросетей в медицине \slnumber}
		Можно выделить три основных направления развития нейросетевых технологий в медицине:
			\begin{enumerate}
				\item Малый набор данных "--- автоматизации сбора и оцифровки медицинских данных
				\item Низкое качество данных "--- увеличение качества данных за счет разработки и внедрения алгоритмов компенсации цветовых искажений
				\item Разметка датасета "--- количественное и качественное увеличение автоматизации разметки данных
			\end{enumerate}
\end{frame}

% Постановка задачи - обязательный слайд
\begin{frame}
    \frametitle{Постановка задач \slnumber}
    \Large
    \begin{enumerate}
		    \item Автоматизация разметки датасета
		 		\item Разработка нейросетевой модели
		 		\item Обучение нейросети с учителем
		 		\item Нивелирование малого количества данных и повышение точности автоматического прогнозирование заболеваний
		 		\item Создание приложения для нужд медицинских работников
    \end{enumerate}
\end{frame}

\begin{frame}
	\frametitle{Стэк технологий \slnumber}
		При разработке приложения был использован следующий стек технологий:
			\begin{enumerate}
			\item Язык программирования Python
			\item Библиотека Tensorflow для разработки и обучения нейросети
			\item cv2 и Pillow для работы с растровой графикой
			\item ImageAI для внедрения нейросетевой модели в приложение
			\item Сервис для рассылки СМС сообщений NotiSend
			\item PySimpleGUI для создания интерфейса приложений
			\end{enumerate}
\end{frame}

\begin{frame}
	\frametitle{Подготовка данных \slnumber}
		Для автоматизации подготовки датасета были написаны следующие скрипты:
		\begin{enumerate}
			\item VideoToImage.py "--- скрипт для преобразования видео в набор фото
			\item ConverterPNG.py "--- скрипт для приведения всех изображений к единому PNG формату
			\item DuplicateRemover.py "--- скрипт дял определения и удаления дубликтов изображений
			\item NormalLabel.py "--- скрипт для автоматической разметки класса <<Нормальная слизистая желудка>>
			\item FolderMaker.py "--- скрипт для распределения данных на тестовые и валидационные наборы
		\end{enumerate}
\end{frame}

\begin{frame}
	\frametitle{Обучение нейросети \slnumber}
		Для обучения нейросети был арендован сервер Google:
		\begin{enumerate}
			\item Два видеоускорителя Nvidia Tesla K80 -- 64гб vRAM
			\item 128гб оперативной памяти
			\item Intel Xeon Gold 6130 -- 16 ядер, 3.7Ггц
			\linebreak
		\end{enumerate}
		
		Характеристики обучения нейросети:
		\begin{enumerate}
			\item 4 класса:
				\begin{itemize}
					\item Застойное содержимое
					\item	Редуцированная кровь
					\item Свежая кровь
					\item Нормальная слизистая желудка
				\end{itemize}
			\item 8300 изображений на класс
			\item Размер батч-файла -- 16
			\item Количество эпох -- 200
		\end{enumerate}
\end{frame}
